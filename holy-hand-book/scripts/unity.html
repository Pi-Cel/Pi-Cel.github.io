Player Movement - Top Down

This script can be used for moving a character in a top down game.

        using UnityEngine;
        // Path of the file should be /Scripts/Player/PlayerMovement.
        namespace Player
        {
        	// Make sure that the game object also has a rigibody.
        	[RequireComponent(typeof(Rigidbody2D))]
        	Public class PlayerMovement : MonoBehaviour
        	{
        		[SerializeField]
        		private float movementSpeed;
        		private Rigidbody2D _rb;

        		private void FixedUpdate()
        		{
        			// First we get the input from the player.
        			var h = Input.GetAxisRaw("Horizontal");
        			var v = Input.GetAxisRaw("Vertical");
        			var inputVector = new Vector2(h, v);

        			// Work out the position to move to.
        			var targetPosition = transform.position + inputVector * movementSpeed;
        			_rb.MovePosition(targetPosition);
        		}
        	}
        }
